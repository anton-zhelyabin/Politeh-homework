git init # Создать новый локальный репозиторий
git clone <url> # Склонировать удаленный репозиторий
git add # Добавить изменения в индекс
git commit # Создать коммит
git push # Отправить изменения на удаленный репозиторий
git pull # Загрузить и слить изменения

Один большой пример:
# Создадим новый локальный репозиторий
git init
# Склонируем удаленный репозиторий
git clone <url>
# Загрузим проект на локальный репозиторий
git pull
# Создадим новую ветку и переключимся на неё
git checkout -b comands
# Создадим новый файл
echo "git comands" > comands.txt
# Добавим его в индекс
git add comands.txt
# Создадим новый коммит
git commit -m "add comands.txt"
# Отправим изменения на удалённый репозиторий
git push -u origin comands
